    <script>
        // URL do Logic App configurada
        const LOGIC_APP_URL = 'https://prod-54.eastus.logic.azure.com:443/workflows/ec5a9a7b95fa4c46962901069af372a8/triggers/When_an_HTTP_request_is_received/paths/invoke?api-version=2016-10-01&sp=%2Ftriggers%2FWhen_an_HTTP_request_is_received%2Frun&sv=1.0&sig=YlRDzSyrRjlf81yqalIUFfiROvRjogvaBWMo_O5m81k';

        let selectedFunctionality = '';
        let lastResult = null;

        // Sele√ß√£o de funcionalidade
        document.querySelectorAll('.functionality-card').forEach(card => {
            card.addEventListener('click', function() {
                document.querySelectorAll('.functionality-card').forEach(c => c.classList.remove('selected'));
                this.classList.add('selected');
                selectedFunctionality = this.dataset.func;
                
                // Mostrar/ocultar campo de pol√≠tica
                const politicaGroup = document.getElementById('politica-group');
                if (selectedFunctionality === 'D') {
                    politicaGroup.style.display = 'block';
                    document.getElementById('conteudo_politica').required = true;
                } else {
                    politicaGroup.style.display = 'none';
                    document.getElementById('conteudo_politica').required = false;
                }
            });
        });

        // Submit do formul√°rio
        document.getElementById('grcForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            // Validar se URL foi configurada
            if (LOGIC_APP_URL === 'COLE_AQUI_A_URL_DO_SEU_LOGIC_APP') {
                alert('‚ö†Ô∏è Erro: Configure a URL do Logic App no c√≥digo JavaScript antes de usar!');
                return;
            }
            
            if (!selectedFunctionality) {
                alert('Por favor, selecione uma funcionalidade.');
                return;
            }

            const formData = {
                funcionalidade: selectedFunctionality,
                framework: document.getElementById('framework').value,
                controle: document.getElementById('controle').value,
                contexto: document.getElementById('contexto').value
            };

            // Adicionar conte√∫do da pol√≠tica se for funcionalidade D
            if (selectedFunctionality === 'D') {
                formData.conteudo_politica = document.getElementById('conteudo_politica').value;
            }

            // Mostrar loading
            document.getElementById('loading').style.display = 'block';
            document.getElementById('results').style.display = 'none';
            document.getElementById('error').style.display = 'none';
            document.querySelector('.submit-btn').disabled = true;

            try {
                console.log('Enviando dados:', formData);

                const response = await fetch(LOGIC_APP_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify(formData)
                });

                if (!response.ok) {
                    throw new Error(`Erro HTTP: ${response.status} - ${response.statusText}`);
                }

                const result = await response.json();
                console.log('Resposta recebida:', result);
                lastResult = result;
                
                // Mostrar resultado formatado
                document.getElementById('resultJson').textContent = JSON.stringify(result, null, 2);
                document.getElementById('results').style.display = 'block';
                
            } catch (error) {
                console.error('Erro completo:', error);
                document.getElementById('error').innerHTML = `
                    <strong>Erro na an√°lise:</strong><br>
                    ${error.message}<br>
                    <small>Verifique o console do navegador (F12) para mais detalhes.</small>
                `;
                document.getElementById('error').style.display = 'block';
            } finally {
                // Ocultar loading
                document.getElementById('loading').style.display = 'none';
                document.querySelector('.submit-btn').disabled = false;
            }
        });

        // Download do resultado
        function downloadResult() {
            if (!lastResult) return;
            
            const filename = `grc-analysis-${selectedFunctionality}-${new Date().toISOString().slice(0,19).replace(/[:.]/g, '-')}.json`;
            const blob = new Blob([JSON.stringify(lastResult, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

        // Verificar configura√ß√£o na inicializa√ß√£o
        window.addEventListener('load', function() {
            if (LOGIC_APP_URL === 'COLE_AQUI_A_URL_DO_SEU_LOGIC_APP') {
                console.warn('‚ö†Ô∏è ATEN√á√ÉO: Configure a URL do Logic App antes de usar!');
            }
        });
